public class Carro {
   
String nome;
String placa;
String chassi;

public String getNome() {
return nome;
}
public void setNome(String nome) {
this.nome = nome;
}
public String getPlaca() {
return placa;
}
public void setPlaca(String placa) {
this.placa = placa;
}
public String getChassi() {
return chassi;
}
public void setChassi(String chassi) {
this.chassi = chassi;
}

}
